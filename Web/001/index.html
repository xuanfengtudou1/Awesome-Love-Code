// ================= 全屏视频播放功能 ================= const fullscreenContainer = document.getElementById('fullscreen-video'); const fullscreenPlayer = document.getElementById('fullscreen-player'); const fullscreenClose = document.getElementById('fullscreen-close'); const fullscreenPlayPause = document.getElementById('fullscreen-playpause'); const fullscreenExit = document.getElementById('fullscreen-exit'); const fullscreenProgress = document.getElementById('fullscreen-progress'); const fullscreenProgressBar = document.getElementById('fullscreen-progress-bar'); const fullscreenBack = document.getElementById('fullscreen-back'); let currentVideoCard = null; // 打开全屏播放器 function openFullscreen(videoElement, cardElement) { if (!videoElement || !fullscreenPlayer) return; currentVideoCard = cardElement; const videoSrc = videoElement.querySelector('source')?.src || videoElement.src; if (videoSrc) { const currentTime = videoElement.currentTime; const isPlaying = !videoElement.paused; fullscreenPlayer.src = videoSrc; fullscreenPlayer.currentTime = currentTime; fullscreenPlayer.load(); fullscreenContainer.classList.add('active'); fullscreenPlayer.play().catch(err => console.log('播放失败:', err)); const playPauseIcon = fullscreenPlayPause.querySelector('i'); playPauseIcon.classList.remove('fa-play'); playPauseIcon.classList.add('fa-pause'); // 暂停原视频 if (isPlaying) { videoElement.pause(); } } } // 关闭全屏播放器 —— 【修复核心】 function closeFullscreen() { if (!fullscreenContainer) return; // 1. 先暂停全屏播放器 fullscreenPlayer.pause(); const currentTime = fullscreenPlayer.currentTime; // 2. 恢复原视频状态（但不自动播放） if (currentVideoCard) { const originalVideo = currentVideoCard.querySelector('video'); if (originalVideo) { originalVideo.currentTime = currentTime; // 【关键】确保原视频保持暂停状态 originalVideo.pause(); // 重置播放图标为「播放」 const originalIcon = currentVideoCard.querySelector('.play-icon-indicator'); const originalStatus = currentVideoCard.querySelector('.video-status'); if (originalIcon) { originalIcon.classList.remove('fa-pause'); originalIcon.classList.add('fa-play'); } if (originalStatus) { originalStatus.style.opacity = '0'; } } } // 3. 隐藏全屏 fullscreenContainer.classList.remove('active'); // 4. 清空视频源，彻底停止播放 setTimeout(() => { fullscreenPlayer.src = ''; fullscreenPlayer.load(); currentVideoCard = null; }, 300); } // 全屏播放/暂停 function toggleFullscreenPlay() { const playPauseIcon = fullscreenPlayPause.querySelector('i'); if (fullscreenPlayer.paused) { fullscreenPlayer.play(); playPauseIcon.classList.remove('fa-play'); playPauseIcon.classList.add('fa-pause'); } else { fullscreenPlayer.pause(); playPauseIcon.classList.remove('fa-pause'); playPauseIcon.classList.add('fa-play'); } } // 更新进度条 function updateProgress() { if (fullscreenPlayer.duration && !isNaN(fullscreenPlayer.duration)) { const percent = (fullscreenPlayer.currentTime / fullscreenPlayer.duration) * 100; fullscreenProgressBar.style.width = percent + '%'; } } // 点击进度条跳转 function seekTo(e) { const rect = fullscreenProgress.getBoundingClientRect(); const clickX = e.clientX - rect.left; const percent = clickX / rect.width; if (fullscreenPlayer.duration) { fullscreenPlayer.currentTime = percent * fullscreenPlayer.duration; } } // 绑定全屏播放器事件 if (fullscreenClose) fullscreenClose.addEventListener('click', closeFullscreen); if (fullscreenBack) fullscreenBack.addEventListener('click', closeFullscreen); if (fullscreenExit) fullscreenExit.addEventListener('click', closeFullscreen); if (fullscreenPlayPause) fullscreenPlayPause.addEventListener('click', toggleFullscreenPlay); if (fullscreenProgress) fullscreenProgress.addEventListener('click', seekTo); if (fullscreenPlayer) { fullscreenPlayer.addEventListener('timeupdate', updateProgress); fullscreenPlayer.addEventListener('ended', () => { const playPauseIcon = fullscreenPlayPause.querySelector('i'); playPauseIcon.classList.remove('fa-pause'); playPauseIcon.classList.add('fa-play'); }); fullscreenPlayer.addEventListener('play', () => { const playPauseIcon = fullscreenPlayPause.querySelector('i'); playPauseIcon.classList.remove('fa-play'); playPauseIcon.classList.add('fa-pause'); }); fullscreenPlayer.addEventListener('pause', () => { const playPauseIcon = fullscreenPlayPause.querySelector('i'); if (fullscreenPlayer.currentTime < fullscreenPlayer.duration) { playPauseIcon.classList.remove('fa-pause'); playPauseIcon.classList.add('fa-play'); } }); }
